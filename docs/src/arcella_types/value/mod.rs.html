10//! Generic value types for Arcella.
11//!
12//! This module defines a universal value representation (`Value`) used across
13//! different parts of the Arcella system (e.g., configuration, ALME protocol, manifests)
14//! to handle structured data in a type-safe manner.
15//!
16//! The `Value` enum provides a flexible way to represent common data types
17//! that can be serialized/deserialized using `serde`.
18
19use serde::{Deserialize, Serialize};
20use std::collections::HashMap;
21
22/// A generic value type that can represent data from configuration, ALME protocol,
23/// WIT interfaces, or other structured sources within the Arcella ecosystem.
24///
25/// This enum serves as a common interchange format for dynamic data, similar to
26/// `serde_json::Value` but tailored for Arcella's specific needs.
27///
28/// It supports:
29/// - Primitive types: `String`, `Integer`, `Float`, `Boolean`, `Null`
30/// - Compound types: `Array` (list of `Value`), `Map` (key-value pairs of `String` to `Value`)
31/// - Error signaling: `Error` (for representing failures during data processing)
32///
33/// # Examples
34///
35/// ```
36/// use arcella_types::value::Value;
37/// use std::collections::HashMap;
38///
39/// // Creating a simple value
40/// let string_val = Value::String("hello".to_string());
41///
42/// // Creating a complex nested structure
43/// let mut map = HashMap::new();
44/// map.insert("key1".to_string(), Value::Integer(42));
45/// map.insert("key2".to_string(), Value::Boolean(true));
46/// let array_val = Value::Array(vec![Value::String("item1".to_string()), Value::Null]);
47/// map.insert("key3".to_string(), array_val);
48/// let complex_val = Value::Map(map);
49/// ```
50#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
51pub enum Value {
52    /// A sequence of `Value`s.
53    Array(Vec<Value>),
54
55    /// A UTF-8 string.
56    String(String),
57
58    /// A signed 64-bit integer.
59    Integer(i64),
60
61    /// A 64-bit floating-point number.
62    Float(f64),
63
64    /// A boolean value.
65    Boolean(bool),
66
67    /// A map of string keys to `Value`s.
68    Map(HashMap<String, Value>),
69
70    /// An explicit null value, representing the absence of data.
71    Null,
72
73    /// An error state, used to signal failures during data processing or conversion.
74    ///
75    /// This variant is useful when a value cannot be correctly parsed or constructed,
76    /// allowing the error to be propagated alongside other valid data.
77    Error(String),
78}
