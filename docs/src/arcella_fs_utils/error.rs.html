10use std::path::PathBuf;
11use thiserror::Error;
12
13/// Result type alias for `arcella-wasmtime` operations.
14pub type Result<T> = std::result::Result<T, ArcellaUtilsError>;
15
16/// Errors that can occur during Wasmtime-to-Arcella conversion.
17#[derive(Error, Debug)]
18pub enum ArcellaUtilsError {
19    /// General-purpose error for unexpected conditions.
20    #[error("Internal error: {0}")]
21    Internal(String),
22
23    /// I/O error (file not found, permission denied, etc.).
24    #[error("I/O error: {0}")]
25    Io(#[from] std::io::Error),
26
27    /// IO error with associated path for better diagnostics
28    #[error("I/O error at {path:?}: {source}")]
29    IoWithPath {
30        source: std::io::Error,
31        path: PathBuf,
32    },
33
34    /// Path not found
35    #[error("Path not found: {path:?}")]
36    PathNotFound {
37        path: PathBuf,
38    },
39
40    /// TOML error
41    #[error("TOML error: {0}")]
42    TOML(String),
43}
