10use std::path::PathBuf;
11
12/// Represents non-critical warnings that can occur during configuration loading.
13///
14/// These warnings are collected during the configuration loading process
15/// when the main logger might not yet be initialized. They are stored
16/// in a buffer and can be processed (e.g., logged) later.
17#[derive(Debug, Clone)]
18pub enum ConfigLoadWarning {
19    /// General-purpose warning for unexpected conditions.
20    Internal(String),
21
22    /// A configuration value was found to be `Null`.
23    NullValueDetected { key: String, file: PathBuf },
24
25    /// An error occurred while processing a configuration value (e.g., unsupported type).
26    ValueError { key: String, error: String, file: PathBuf },
27
28    /// A configuration file was included more than once (cycle or duplicate).
29    DuplicateInclude { path: PathBuf, included_from: PathBuf },
30
31    /// A configuration file was retried for processing (e.g., due to internal logic or depth limits).
32    RetriedProcessing { path: PathBuf },
33
34    /// A file specified in `includes` was skipped because it did not pass validation
35    /// (e.g., not a `.toml` file or is a `.template.toml` file).
36    SkippedInvalidFile { path: PathBuf },
37
38    /// An unknown TOML type was encountered that could not be converted.
39    UnknownTomlType { key: String, type_name: String, file: PathBuf },
40
41    /// Represents a warning that occurs when the maximum depth of includes is reached.
42    MaxDepthReached { path: PathBuf },
43}
44
45impl std::fmt::Display for ConfigLoadWarning {
46    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
47        match self {
48            ConfigLoadWarning::Internal(msg) => {
49                write!(f, "{}", msg)
50            }
51            ConfigLoadWarning::NullValueDetected { key, file } => {
52                write!(f, "Null value found for key '{}' in file {:?}", key, file)
53            }
54            ConfigLoadWarning::ValueError { key, error, file } => {
55                write!(f, "Error processing value for key '{}' in file {:?}: {}", key, file, error)
56            }
57            ConfigLoadWarning::DuplicateInclude { path, included_from } => {
58                write!(f, "Duplicate include path '{:?}' found, already included from {:?}", path, included_from)
59            }
60            ConfigLoadWarning::RetriedProcessing { path } => {
61                write!(f, "Retried processing of config file {:?}", path)
62            }
63            ConfigLoadWarning::SkippedInvalidFile { path } => {
64                write!(f, "Skipped invalid file in includes: {:?}", path)
65            }
66            ConfigLoadWarning::UnknownTomlType { key, type_name, file } => {
67                write!(f, "Unknown TOML type '{}' for key '{}' in file {:?}", type_name, key, file)
68            }
69            ConfigLoadWarning::MaxDepthReached { path } => {
70                write!(f, "Maximum include depth reached for file {:?}", path)
71            }
72        }
73    }
74}
