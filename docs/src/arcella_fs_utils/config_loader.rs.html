10//! Recursive configuration loader for Arcella.
11//!
12//! This module provides functions for loading TOML-based configuration files,
13//! including support for recursive inclusion of other files via the `includes` key.
14//! It handles circular dependencies, limits recursion depth, and collects warnings
15//! during the loading process for later reporting.
16
17//!
18//! ## Key Features
19//!
20//! - **File inclusion**: Files can include others via `includes = "file.toml"` or `includes = ["a.toml", "b.toml"]`.
21//! - **Glob and directory support**: `includes` may contain glob patterns (e.g., `"config.d/*.toml"`) or directories.
22//! - **Cycle detection**: Each file is loaded at most once across the entire configuration tree.
23//! - **Depth limiting**: Prevents infinite recursion due to deep or cyclic includes (`MAX_CONFIG_DEPTH = 5`).
24//! - **Warning collection**: Non-fatal issues (e.g., pruned subtrees, duplicate includes) are collected for later inspection.
25//! - **Deterministic ordering**: Included files from globs are sorted lexicographically to ensure consistent behavior.
26
27//!
28//! ## Path Resolution
29//!
30//! All paths in `includes` are resolved **relative to `ConfigLoadParams::config_dir`**,  
31//! *not* relative to the including file. This ensures predictable and reproducible behavior
32//! regardless of the inclusion chain.
33//!
34//! ## Missing Files
35//!
36//! If a path in `includes` does not exist or is not a valid TOML file (e.g., a `.template.toml` file),
37//! it is **silently skipped** and a `SkippedInvalidFile` warning is recorded.
38//! This allows optional configuration files (e.g., `local.toml`) to be absent without causing an error.
39

51/// The maximum allowed recursion depth when loading configuration files.
52///
53/// This prevents stack overflow or excessive resource consumption from deeply nested
54/// or circular `includes`. The root file is at depth 0, so up to `MAX_CONFIG_DEPTH + 1`
55/// files can be loaded in a single inclusion chain.
56///
57/// Example: with `MAX_CONFIG_DEPTH = 5`, the following is allowed:
58/// `root.toml → a.toml → b.toml → c.toml → d.toml → e.toml` (6 files total).
59/// Attempting to include a 7th file will trigger a `MaxDepthReached` warning and skip loading.
60const MAX_CONFIG_DEPTH: usize = 5;
61
62/// Resolves a list of include patterns into absolute file paths.
63///
64/// Relative paths in the `includes` list are resolved relative to the `parent_dir`.
65/// Absolute paths are kept as is.
66///
67/// # Arguments
68///
69/// * `includes` - A vector of string patterns representing file or directory paths.
70/// * `parent_dir` - The base directory to resolve relative paths against.
71///
72/// # Returns
73///
74/// A `Result` containing a `HashSet` of resolved `PathBuf`s or an error.
75pub fn resolve_include_paths(
76    includes: &[String],
77    parent_dir: &Path
78) -> ArcellaUtilsResult<HashSet<PathBuf>> {
79    let mut all_paths = HashSet::new();
80    for include_pattern in includes {
81        let pattern_path = PathBuf::from(include_pattern);
82        if pattern_path.is_absolute() {
83            // If the path is absolute, leave it as is.
84            all_paths.insert(pattern_path);
85        } else {
86            // If relative, make it relative to config_dir
87            all_paths.insert(parent_dir.join(include_pattern));
88        }
89    }
90    Ok(all_paths)
91}

94/// Recursively loads configuration files starting from `config_file_path`, including files specified in `includes`.
95///
96/// This function reads the initial configuration file, parses it, extracts `includes`,
97/// and then recursively processes those included files up to `MAX_CONFIG_DEPTH`.
98/// It collects both configuration data and non-critical warnings during the process.
99///
100/// Warnings are collected in the provided `state.warnings` vector. This allows reporting issues
101/// (like null values, duplicate includes, etc.) that occur before the main logger is initialized.
102///
103/// Critical errors (like I/O failures, TOML syntax errors) will stop the loading process
104/// and return an `Err`.
105///
106/// # Arguments
107///
108/// * `params` – Immutable loading parameters (prefix, config directory).
109/// * `state` – Mutable state tracking visited files, loaded files, and warnings.
110/// * `config_file_path` – The absolute or relative path to the configuration file to load.
111/// * `included_from` – The file that included `config_file_path` (for cycle diagnostics).
112/// * `current_depth` – Current inclusion depth (0 for the root file).
113///
114/// # Returns
115///
116/// A `Result` containing a `Vec<TomlFileData>` representing the loaded configurations,
117/// or an `ArcellaUtilsError` if a critical error occurs.
118///
119/// Note: Each file is loaded at most once globally (not per inclusion path), to avoid redundant work
120/// and ensure deterministic behavior.
121pub async fn load_config_recursive(
122    params: &ConfigLoadParams,
123    state: &mut ConfigLoadState,
124    config_file_path: &Path,
125    included_from: Option<&Path>,
126    current_depth: usize,
127) -> ArcellaUtilsResult<Vec<TomlFileData>> {
128    // Enforce maximum inclusion depth
129    if current_depth > MAX_CONFIG_DEPTH {
130        state.warnings.push(ConfigLoadWarning::MaxDepthReached {
131            path: config_file_path.to_path_buf(),
132        });
133        return Ok(vec![]); // Reached maximum depth
134    }
135
136    // Prevent loading the same file more than once (global deduplication)
137    if state.visited_paths.contains(config_file_path) {
138        state.warnings.push(ConfigLoadWarning::DuplicateInclude {
139            path: config_file_path.to_path_buf(),
140            included_from: included_from.map(|p| p.to_path_buf())
141                .unwrap_or_else(|| config_file_path.to_path_buf()),
142        });
143        return Ok(vec![]); // Not an error, just break the recursion
144    }
145
146    // Read file content first; only mark as visited after successful read
147    // to avoid poisoning the state on transient I/O errors.    
148
149    let content = tokio::fs::read_to_string(config_file_path)
150        .await
151        .map_err(|e| ArcellaUtilsError::IoWithPath {
152            source: e,
153            path: config_file_path.to_path_buf(),
154        })?;
155
156    // Now it's safe to mark the file as visited
157    state.visited_paths.insert(config_file_path.to_path_buf());
158    let (file_idx, _) = state.config_files.insert_full(config_file_path.to_path_buf());
159
160    let all_configs = load_config_recursive_from_content(
161        params,
162        state,
163        &content,
164        file_idx,
165        config_file_path,
166        current_depth,
167    ).await?;
168
169    // visited_paths.remove(config_file_path); // Optional, if cycles are checked only within one traversal path
170
171    Ok(all_configs)
172}

174/// Parses configuration content and recursively loads included files.
175///
176/// This function parses the provided TOML `content`, extracts `includes`,
177/// resolves them relative to `params.config_dir`, and recursively loads each included file.
178/// Globs and directories in `includes` are expanded and sorted lexicographically.
179///
180/// # Arguments
181///
182/// * `params` – Immutable loading parameters.
183/// * `state` – Mutable loading state.
184/// * `content` – Raw TOML content of the current file.
185/// * `file_idx` – Unique index of this file (for value provenance).
186/// * `config_file_path` – Path of the current file (used for diagnostics).
187/// * `current_depth` – Current inclusion depth.
188///
189/// # Returns
190///
191/// A vector of `TomlFileData` for this file and all recursively included files.
192pub async fn load_config_recursive_from_content(
193    params: &ConfigLoadParams,
194    state: &mut ConfigLoadState,
195    content: &str,
196    file_idx: usize,
197    config_file_path: &Path,
198    current_depth: usize,
199) -> ArcellaUtilsResult<Vec<TomlFileData>> {
200
201    let (config, result) = toml::parse_and_collect(&content, &params.prefix, file_idx)?;
202    if result == TraversalResult::Pruned{
203        state.warnings.push(ConfigLoadWarning::Pruned {
204            path: config_file_path.to_path_buf(),
205        });
206    }
207
208     // --- Check values for Null or other issues (example) ---
209    // This could be extracted into a separate function for checking TomlFileData
210    for (key, (value, _)) in &config.values {
211        if matches!(value, TomlValue::Null) {
212            state.warnings.push(ConfigLoadWarning::NullValueDetected {
213                key: key.clone(),
214                file: config_file_path.to_path_buf(),
215            });
216        }
217    }
218
219    // Resolve and expand includes (e.g., globs, directories) into concrete file paths.
220    // The result is sorted lexicographically to ensure deterministic loading order.
221    // Invalid or missing paths are skipped and recorded as warnings.
222    let include_paths = collect_toml_includes(
223        &config.includes, 
224        &params.config_dir, 
225        &mut state.warnings,
226    ).await?;
227
228    let mut all_configs = vec![config];
229
230    // Recursively load each included file
231    for include_path in include_paths {
232        // Pin the future returned by the recursive call
233        let sub_configs_future = Box::pin(load_config_recursive(
234            params,
235            state,
236            &include_path,
237            Some(config_file_path),
238            current_depth + 1,
239        ));
240        // Await the pinned future
241        let mut sub_configs = sub_configs_future.await?;
242        all_configs.append(&mut sub_configs);
243    };
244
245    Ok(all_configs)
246}

248/// Loads configuration files recursively starting from a single file.
249///
250/// This is a convenience entry point that initiates recursive loading from a root file.
251/// It does not initialize `state`; the caller must provide a fresh or reused `ConfigLoadState`.
252///
253/// # Returns
254///
255/// A `Result` containing a `Vec<TomlFileData>` with all loaded configuration data.
256/// Warnings are accumulated in `state.warnings` and should be inspected by the caller.
257pub async fn load_config_recursive_from_file(
258    params: &ConfigLoadParams,
259    state: &mut ConfigLoadState,
260    config_file_path: &Path,
261) -> ArcellaUtilsResult<Vec<TomlFileData>> {
262
263    load_config_recursive(
264        params,
265        state,
266        config_file_path, 
267        None,
268        0,
269    ).await
270
271}

417            config_files: IndexSet::new(),
418            visited_paths: HashSet::new(),

421
422        // Create a chain of files that exceeds MAX_CONFIG_DEPTH

424            let current_file = config_dir.join(format!("level_{}.toml", i));
425            let mut content = format!("key_{} = {}", i, i);

427            fs::write(&current_file, content).unwrap();
428        }
429
430        let root_file = config_dir.join("level_0.toml");
431
432        let params = ConfigLoadParams {
433            prefix: vec!["arcella".to_string()],
434            config_dir: config_dir.to_path_buf(),
435        };
436
437        let configs = load_config_recursive_from_file(
438            &params,
439            &mut state,
440            &root_file,
441        ).await.unwrap();
442
443        // Should stop after MAX_CONFIG_DEPTH
444        // The exact number of loaded configs might vary slightly depending on implementation details,
445        // but the key point is that it stops and generates a warning.
