10use indexmap::IndexSet;
11use std::collections::HashSet;
12use std::path::PathBuf;
13
14use arcella_types::config::ConfigValues;
15use crate::ConfigLoadWarning;

17/// Maximum allowed recursion depth when traversing nested TOML tables.
18///
19/// Prevents stack overflow due to deeply nested or maliciously crafted TOML.
20/// This limit applies only to table nesting, not array depth or file inclusion depth
21/// (which is controlled by `MAX_CONFIG_DEPTH` in `config_loader.rs`).
22pub const MAX_TOML_DEPTH: usize = 10;
23
24/// Template file suffix
25pub const TEMPLATE_TOML_SUFFIX: &str = ".template.toml";
26
27// Immutable parameters — can be freely cloned
28#[derive(Debug, Clone)]
29pub struct ConfigLoadParams {
30    pub prefix: Vec<String>,
31    pub config_dir: PathBuf,
32}
33
34// Mutable state — passed by &mut reference
35pub struct ConfigLoadState {
36    /// All configuration files that have been successfully loaded, in order of inclusion.
37    pub config_files: IndexSet<PathBuf>,
38
39    /// Tracks files currently in the inclusion stack to detect cyclic includes.
40    pub visited_paths: HashSet<PathBuf>,
41
42    /// Non-fatal warnings collected during loading.
43    pub warnings: Vec<ConfigLoadWarning>,
44}
45
46impl Default for ConfigLoadState {
47    fn default() -> Self {
48        Self {
49            config_files: IndexSet::new(),
50            visited_paths: HashSet::new(),
51            warnings: Vec::new(),
52        }
53    }
54}
55
56#[derive(Debug, Clone, PartialEq)]
57pub struct TomlFileData {
58    pub includes: Vec<String>,
59    pub values: ConfigValues,
60}
61
62/// Indicates the outcome of a recursive traversal of a TOML document.
63///
64/// - `Full`: The entire subtree was processed without hitting depth limits.
65/// - `Pruned`: Traversal was stopped early because `MAX_TOML_DEPTH` was exceeded.
66///   This is a non-fatal condition; a warning is issued, but loading continues.
67#[derive(Debug, Clone, PartialEq)]
68pub enum TraversalResult { Full, Pruned }
