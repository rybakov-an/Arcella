23/// Maximum allowed length of an incoming ALME request in bytes.
24/// Requests exceeding this limit will be rejected to prevent resource exhaustion.
25static MAX_REQUEST_LENGTH: usize = 64 * 1024; // 64 KB
26
27static MAX_READER_TIMEOUT: u64 = 60; // seconds

29/// Spawns the ALME (Arcella Local Management Extensions) server as a background task.
30///
31/// The server listens on a Unix domain socket at the specified `socket_path` and handles
32/// incoming management commands (e.g., `install`, `start`, `status`) by delegating them
33/// to the provided shared `ArcellaRuntime` instance.
34///
35/// On startup, any existing file at `socket_path` is removed to handle stale sockets.
36/// The socket file is created with permissions `0o600` (read/write for owner only) for security.
37///
38/// A graceful shutdown can be initiated by calling [AlmeServerHandle::shutdown],
39/// which signals the server to stop accepting new connections, notifies all active connection 
40/// handlers to terminate, and removes the Unix socket file once the server loop exits.
41///  
42/// # Arguments
43///
44/// * `socket_path` - The filesystem path where the Unix socket will be created.
45/// * `runtime` - A thread-safe shared reference to the main Arcella runtime instance.
46///
47/// # Returns
48///
49/// An `AlmeServerHandle` that can be used to initiate a graceful shutdown of the server.
50///
51/// # Errors
52///
53/// Returns an error if:
54/// - The socket cannot be bound (e.g., due to permission issues).
55/// - The socket file permissions cannot be set
56pub async fn spawn_server(
57    socket_path: PathBuf, 
58    runtime: Arc<RwLock<ArcellaRuntime>>,
59) -> ArcellaResult<super::AlmeServerHandle> {
60
61    if socket_path.exists() {
62        if let Err(e) = fs::remove_file(&socket_path) {
63            eprintln!("[ALME] Warning: failed to remove stale socket {:?}: {}", socket_path, e);
64        }
65    }
66
67    let listener = UnixListener::bind(&socket_path)?;
68    fs::set_permissions(&socket_path, fs::Permissions::from_mode(0o600))?;
69
70    let (shutdown_tx, mut shutdown_rx) = broadcast::channel::<()>(1);
71
72    let socket_path_clone = socket_path.clone();
73    let runtime_clone = runtime.clone();
74    let join_handle = tokio::spawn(async move {
75        let result = run_server_loop(listener, runtime_clone, shutdown_rx).await;
76
77        // Remove socket on shutdown
78        if let Err(e) = fs::remove_file(&socket_path_clone) {
79            eprintln!("[ALME] Failed to remove ALME socket {:?}: {}", socket_path_clone, e);
80        }
81
82        result
83    });
84
85    Ok(super::AlmeServerHandle {
86        shutdown_tx: Some(shutdown_tx),
87        join_handle: Some(join_handle),
88    })
89}

91/// Runs the main accept loop for the ALME server.
92///
93/// This function continuously accepts new incoming Unix socket connections
94/// until a shutdown signal is received via the `shutdown_rx` channel.
95/// For each connection, it spawns a dedicated asynchronous task to handle
96/// the client's requests via [`handle_connection`].
97///
98/// The loop is resilient to transient client or I/O errors but will exit
99/// on listener errors or explicit shutdown.
100/// 
101/// # Arguments
102///
103/// * `listener` - The bound `UnixListener` to accept connections from.
104/// * `runtime` - Shared access to the Arcella runtime for command execution.
105/// * `shutdown_rx` - Receiver for shutdown signals.
106async fn run_server_loop(
107    listener: UnixListener,
108    runtime: Arc<RwLock<ArcellaRuntime>>,
109    mut shutdown_rx: broadcast::Receiver<()>,
110) -> ArcellaResult<()> {
111
112    loop {
113        tokio::select! {
114            accept_result = listener.accept() => {
115                match accept_result {
116                    Ok((stream, _)) => {
117                        let rt = runtime.clone();
118                        let mut shutdown_rx_clone = shutdown_rx.resubscribe();
119                        tokio::spawn(async move {
120                            if let Err(e) = handle_connection(stream, rt, shutdown_rx_clone).await {
121                                eprintln!("[ALME] Connection handler error: {:?}", e);
122                            }
123                        });
124                    },
125                    Err(e) => {
126                        eprintln!("[ALME] Listener accept error: {:?}", e);
127                        break;
128                    }
129                }
130            }
131            _ = shutdown_rx.recv() => {
132                eprintln!("[ALME] Listener received shutdown signal");
133                break;
134            }
135        }
136    }
137
138    Ok(())
139												   
140}

142/// Handles a single ALME client connection for its entire lifetime.
143///
144/// This function runs a loop that:
145/// 1. Reads line-oriented JSON commands from the client (one per line),
146/// 2. Skips empty or whitespace-only lines,
147/// 3. Parses each line as an [`AlmeRequest`],
148/// 4. Dispatches the request to the Arcella runtime,
149/// 5. Sends back a JSON-encoded [`AlmeResponse`].
150///
151/// The connection remains open until one of the following occurs:
152/// - The client closes the connection (EOF),
153/// - A read/write I/O error occurs,
154/// - A global shutdown signal is received via `shutdown_rx`.
155///
156/// Empty lines are ignored (no response is sent).
157/// 
158/// # Arguments
159///
160/// * `stream` - The connected Unix stream to communicate with the client.
161/// * `runtime` - Shared access to the Arcella runtime for executing commands.
162/// * `shutdown_rx` - Receiver for global shutdown signals.
163async fn handle_connection(
164    stream: UnixStream, 
165    runtime: Arc<RwLock<ArcellaRuntime>>,
166    mut shutdown_rx: broadcast::Receiver<()>,
167) -> ArcellaResult<()> {
168
169    let (reader, mut writer) = tokio::io::split(stream);
170    let mut reader = BufReader::new(reader);
171    let mut buffer = String::new();
172
173    let result = loop {
174        buffer.clear();
175
176        let line = tokio::select! {
177            reader_result = timeout(TokioDuration::from_secs(MAX_READER_TIMEOUT), reader.read_line(&mut buffer)) => {
178                match reader_result {
179                    Ok(Ok(0)) => {
180                        break Ok(()); // EOF - client close connection
181                    },
182                    Ok(Ok(n)) => {
183                        if n > MAX_REQUEST_LENGTH {
184                            let resp = AlmeResponse::error("Request too large");
185                            send_response(&mut writer, &resp).await?;
186                            continue;
187                        }
188                        let trimmed = buffer.trim_end_matches(&['\r', '\n']).trim();
189                        if trimmed.is_empty() {
190                            continue;
191                        }
192                        trimmed.to_string()
193                    },
194                    Ok(Err(e)) => {
195                        eprintln!("[ALME] Recieved error: {}", e);
196                        return Err(ArcellaError::Io(e));
197                    },
198                    _ => {
199                        eprintln!("[ALME] Reader timeout");
200                        let _ = writer.shutdown().await;
201                        return Ok(());
202                    }
203                }
204            },
205            _ = shutdown_rx.recv() => {
206                eprintln!("[ALME] Connection handler received shutdown signal");
207                let _ = writer.shutdown().await;
208                return Ok(());
209            },
210        };
211
212        let request: AlmeRequest = match serde_json::from_str(&line) {
213            Ok(req) => req,
214            Err(e) => {
215                let message = format!("Invalid JSON: {}", e);
216                let resp = AlmeResponse::error(&message);
217                send_response(&mut writer, &resp).await?;
218                eprintln!("[ALME] {message}");
219                continue;
220            }
221        };
222
223        let response = match request {
224            AlmeRequest::Ping => AlmeResponse {
225                success: true,
226                message: "pong".to_string(),
227                data: None,
228            },
229            AlmeRequest::Status => {
230                let runtime_guard = runtime.read().await;
231
232                let runtime_status = runtime_guard.status()?;
233
234                let start_time: OffsetDateTime = runtime_status.start_time.into();
235                let start_time_rfc3339 = start_time.format(&time::format_description::well_known::Rfc3339).unwrap();
236
237                let data = serde_json::json!({
238                    "version": env!("CARGO_PKG_VERSION"),
239                    "pid": runtime_status.pid,
240                    "start_time": format!("{}", start_time_rfc3339),
241                    "uptime": runtime_status.uptime.as_secs(),
242                    "socket_path": runtime_guard.config.socket_path.to_string_lossy(),
243                    "worker_groups": "",
244                    "modules": "",
245                });
246
247                AlmeResponse {
248                    success: true,
249                    message: "Arcella runtime is active".to_string(),
250                    data: Some(data),
251                }
252            },
253            AlmeRequest::ListModules => AlmeResponse {
254                success: true,
255                message: "No modules".to_string(),
256                data: Some(serde_json::json!([])),
257            },
258        };
259
260        send_response(&mut writer, &response).await?;
261
262    };
263
264    result
265
266}

268/// Serializes an [`AlmeResponse`] to JSON and writes it to the client stream.
269///
270/// A newline (`\n`) is appended to ensure line-oriented parsing on the client side.
271/// If the write fails (e.g., because the client disconnected), the error is returned
272/// so the connection handler can terminate gracefully.
273/// 
274/// # Arguments
275///
276/// * `stream` - The writable half of the Unix stream to send the response to.
277/// * `response` - The response object to serialize and send.
278async fn send_response(
279    stream: &mut WriteHalf<UnixStream>,
280    response: &AlmeResponse,
281) -> ArcellaResult<()> {
282    let mut json = serde_json::to_vec(response)
283        .map_err(|e| ArcellaError::Serialization(e.to_string()))?;
284    json.push(b'\n');
285    let _ = stream.write_all(&json).await.map_err(|e| {
286        eprintln!("[ALME] Failed to send response: {}", e);
287        ArcellaError::Io(e)
288    });
289    Ok(())
290}

489
490        // Command 2: Status
491        writer.write_all(b"{\"cmd\":\"Status\"}\n").await.unwrap();
492        writer.flush().await.unwrap();
493
494        reader.read_line(&mut response_line).await.unwrap();
495        let resp2: AlmeResponse = serde_json::from_str(&response_line).unwrap();
496        response_line.clear();
497
498        // Command 3: ListModules
499        writer.write_all(b"{\"cmd\":\"ListModules\"}\n").await.unwrap();
500        writer.flush().await.unwrap();
501
502        reader.read_line(&mut response_line).await.unwrap();
503        let resp3: AlmeResponse = serde_json::from_str(&response_line).unwrap();
504
505        assert!(resp1.success);
506        assert_eq!(resp1.message, "pong");
507        assert!(resp2.success);
508        assert_eq!(resp2.message, "Arcella runtime is active");
509        assert!(resp2.data.is_some());
510        assert!(resp3.success);
511        assert!(resp3.data.is_some());

513        
514        // Close socket
515        drop(writer);
516        drop(reader);
517
518        alme_handle.shutdown().await.unwrap();
519
520        assert_eq!(modules.len(), 0); // пока пусто
