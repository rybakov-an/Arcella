10use std::sync::Arc;
11use std::path::PathBuf;
12
13use crate::config::ArcellaConfig;
14use crate::error::{ArcellaError, Result as ArcellaResult};
15
16pub struct StorageManager {
17    pub base_dir: PathBuf,
18    pub config_dir: PathBuf,
19    pub modules_dir: PathBuf,
20    pub cache_dir: PathBuf,
21}
22
23impl StorageManager {
24    pub async fn new(
25        config: &Arc<ArcellaConfig>,
26    ) -> ArcellaResult<Self> {
27
28        let base_dir = config.base_dir.clone();//.unwrap_or_else(|| PathBuf::from("."));
29        let config_dir = config.config_dir.clone();//.unwrap_or_else(|| base_dir.join("config"));
30        let modules_dir = config.modules_dir.clone();//.unwrap_or_else(|| base_dir.join("modules"));
31        let cache_dir = config.cache_dir.clone();//.unwrap_or_else(|| base_dir.join("cache"));
32
33        let manager = Self {
34            base_dir,
35            config_dir,
36            modules_dir,
37            cache_dir,
38        };
39
40        manager.ensure_directories().await?;
41        Ok(manager)
42
43    }
44
45    async fn ensure_directories(&self) -> ArcellaResult<()> {
46        if !self.base_dir.exists() {
47            tokio::fs::create_dir_all(&self.base_dir).await?;
48            tracing::info!("Created base directory: {:?}", self.base_dir);
49            #[cfg(unix)]
50            {
51                use std::os::unix::fs::PermissionsExt;
52                let mut perms = tokio::fs::metadata(&self.base_dir).await?.permissions();
53                perms.set_mode(0o700);
54                tokio::fs::set_permissions(&self.base_dir, perms).await?;
55                tracing::info!("Set permissions for base directory: {:?}", self.base_dir);
56            }
57        }
58
59        if !self.config_dir.exists() {
60            tokio::fs::create_dir_all(&self.config_dir).await?;
61            tracing::info!("Created config directory: {:?}", self.config_dir);
62        }
63
64        if !self.modules_dir.exists() {
65            tokio::fs::create_dir_all(&self.modules_dir).await?;
66            tracing::info!("Created modules directory: {:?}", self.modules_dir);
67        }
68
69        if !self.cache_dir.exists() {
70            tokio::fs::create_dir_all(&self.cache_dir).await?;
71            tracing::info!("Created cache directory: {:?}", self.cache_dir);
72        }
73
74        Ok(())
75    } 
76
77}
