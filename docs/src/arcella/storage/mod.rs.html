10use std::sync::Arc;
11use std::path::PathBuf;
12
13use crate::config::ArcellaConfig;
14use crate::error::{ArcellaError, Result as ArcellaResult};
15
16pub struct StorageManager {
17    pub base_dir: PathBuf,
18    pub config_dir: PathBuf,
19    pub modules_dir: PathBuf,
20    pub cache_dir: PathBuf,
21}
22
23impl StorageManager {
24    pub async fn new(
25        config: &Arc<ArcellaConfig>,
26    ) -> ArcellaResult<Self> {
27        let manager = Self {
28            base_dir: config.base_dir.clone(),
29            config_dir: config.config_dir.clone(),
30            modules_dir: config.modules_dir.clone(),
31            cache_dir: config.cache_dir.clone(),
32        };
33
34        manager.ensure_directories().await?;
35        Ok(manager)
36
37    }
38
39    async fn ensure_directories(&self) -> ArcellaResult<()> {
40        if !self.base_dir.exists() {
41            std::fs::create_dir_all(&self.base_dir)?;
42            #[cfg(unix)]
43            {
44                use std::os::unix::fs::PermissionsExt;
45                let mut perms = std::fs::metadata(&self.base_dir)?.permissions();
46                perms.set_mode(0o700);
47                std::fs::set_permissions(&self.base_dir, perms)?;
48            }
49        }
50
51        if !self.config_dir.exists() {
52            std::fs::create_dir_all(&self.config_dir)?;
53        }
54
55        if !self.modules_dir.exists() {
56            std::fs::create_dir_all(&self.modules_dir)?;
57        }
58
59        if !self.cache_dir.exists() {
60            std::fs::create_dir_all(&self.cache_dir)?;
61        }
62
63        Ok(())
64
65    } 
66
67}
