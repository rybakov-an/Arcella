10//! ALME (Arcella Local Management Extensions) Unix socket server implementation.
11//!
12//! This module provides the core IPC server that enables external tools
13//! (such as the CLI, monitoring agents, or scripts) to interact with the
14//! Arcella runtime daemon via a secure, local Unix domain socket.
15//!
16//! The server:
17//! - Listens on a filesystem socket (e.g., `~/.arcella/alme`) with `0o600` permissions
18//! - Accepts line-oriented JSON requests (one command per line)
19//! - Dispatches commands to handlers in [`crate::alme::commands`]
20//! - Returns structured JSON responses
21//! - Supports graceful shutdown via a broadcast channel
22//! - Enforces security limits (max request size, read timeout)
23//! - Automatically cleans up stale socket files on startup
24//!
25//! The protocol is synchronous and connection-scoped: each client may send
26//! multiple commands over a single connection, and the server responds to each
27//! in order. The server is designed for local administration only and is not
28//! intended for network exposure.
29

44/// Maximum allowed length of an incoming ALME request in bytes.
45/// Requests exceeding this limit will be rejected to prevent resource exhaustion.
46static MAX_REQUEST_LENGTH: usize = 64 * 1024; // 64 KB
47
48static MAX_READER_TIMEOUT: u64 = 60; // seconds

50/// Spawns the ALME (Arcella Local Management Extensions) server as a background task.
51///
52/// The server listens on a Unix domain socket at the specified `socket_path` and handles
53/// incoming management commands (e.g., `install`, `start`, `status`) by delegating them
54/// to the provided shared `ArcellaRuntime` instance.
55///
56/// On startup, any existing file at `socket_path` is removed to handle stale sockets.
57/// The socket file is created with permissions `0o600` (read/write for owner only) for security.
58///
59/// A graceful shutdown can be initiated by calling [crate::alme::AlmeServerHandle::shutdown],
60/// which signals the server to stop accepting new connections, notifies all active connection 
61/// handlers to terminate, and removes the Unix socket file once the server loop exits.
62///  
63/// # Arguments
64///
65/// * `socket_path` - The filesystem path where the Unix socket will be created.
66/// * `runtime` - A thread-safe shared reference to the main Arcella runtime instance.
67///
68/// # Returns
69///
70/// An `AlmeServerHandle` that can be used to initiate a graceful shutdown of the server.
71///
72/// # Errors
73///
74/// Returns an error if:
75/// - The socket cannot be bound (e.g., due to permission issues).
76/// - The socket file permissions cannot be set
77pub async fn spawn_server(
78    socket_path: PathBuf, 
79    runtime: Arc<RwLock<ArcellaRuntime>>,
80) -> ArcellaResult<super::AlmeServerHandle> {
81
82    if socket_path.exists() {
83        if let Err(e) = fs::remove_file(&socket_path) {
84            tracing::error!("Failed to remove stale socket {:?}: {}", socket_path, e);
85        }
86    }
87
88    let listener = UnixListener::bind(&socket_path)?;
89    tracing::debug!("Bind ALME server to socket: {:?}", socket_path);
90    fs::set_permissions(&socket_path, fs::Permissions::from_mode(0o600))?;
91    tracing::debug!("Set permissions on ALME socket: {:?}", socket_path);
92
93    let (shutdown_tx, shutdown_rx) = broadcast::channel::<()>(1);
94
95    let socket_path_clone = socket_path.clone();
96    let runtime_clone = runtime.clone();
97    let join_handle = tokio::spawn(async move {
98        let result = run_server_loop(listener, runtime_clone, shutdown_rx).await;
99
100        // Remove socket on shutdown
101        if let Err(e) = fs::remove_file(&socket_path_clone) {
102            tracing::error!("Failed to remove ALME socket {:?}: {}", socket_path_clone, e);
103        }
104
105        result
106    });
107
108    Ok(super::AlmeServerHandle {
109        shutdown_tx: Some(shutdown_tx),
110        join_handle: Some(join_handle),
111    })
112}

114/// Runs the main accept loop for the ALME server.
115///
116/// This function continuously accepts new incoming Unix socket connections
117/// until a shutdown signal is received via the `shutdown_rx` channel.
118/// For each connection, it spawns a dedicated asynchronous task to handle
119/// the client's requests via [`handle_connection`].
120///
121/// The loop is resilient to transient client or I/O errors but will exit
122/// on listener errors or explicit shutdown.
123/// 
124/// # Arguments
125///
126/// * `listener` - The bound `UnixListener` to accept connections from.
127/// * `runtime` - Shared access to the Arcella runtime for command execution.
128/// * `shutdown_rx` - Receiver for shutdown signals.
129async fn run_server_loop(
130    listener: UnixListener,
131    runtime: Arc<RwLock<ArcellaRuntime>>,
132    mut shutdown_rx: broadcast::Receiver<()>,
133) -> ArcellaResult<()> {
134    tracing::info!("Starting ALME server listener");
135
136    loop {
137        tokio::select! {
138            accept_result = listener.accept() => {
139                match accept_result {
140                    Ok((stream, _addr)) => {
141                        tracing::info!("Get new connection");
142                        let rt = runtime.clone();
143                        let shutdown_rx_clone = shutdown_rx.resubscribe();
144                        tokio::spawn(async move {
145                            if let Err(e) = handle_connection(stream, rt, shutdown_rx_clone).await {
146                                tracing::error!("Connection handler error: {:?}", e);
147                            }
148                        });
149                    },
150                    Err(e) => {
151                        tracing::error!("Listener accept error: {:?}", e);
152                        break;
153                    }
154                }
155            }
156            _ = shutdown_rx.recv() => {
157                tracing::debug!("Listener received shutdown signal");
158                break;
159            }
160        }
161    }
162
163    Ok(())
164												   
165}

167/// Handles a single ALME client connection for its entire lifetime.
168///
169/// This function runs a loop that:
170/// 1. Reads line-oriented JSON commands from the client (one per line),
171/// 2. Skips empty or whitespace-only lines,
172/// 3. Parses each line as an [`AlmeRequest`],
173/// 4. Dispatches the request to the Arcella runtime,
174/// 5. Sends back a JSON-encoded [`AlmeResponse`].
175///
176/// The connection remains open until one of the following occurs:
177/// - The client closes the connection (EOF),
178/// - A read/write I/O error occurs,
179/// - A global shutdown signal is received via `shutdown_rx`.
180///
181/// Empty lines are ignored (no response is sent).
182/// 
183/// # Arguments
184///
185/// * `stream` - The connected Unix stream to communicate with the client.
186/// * `runtime` - Shared access to the Arcella runtime for executing commands.
187/// * `shutdown_rx` - Receiver for global shutdown signals.
188async fn handle_connection(
189    stream: UnixStream, 
190    runtime: Arc<RwLock<ArcellaRuntime>>,
191    mut shutdown_rx: broadcast::Receiver<()>,
192) -> ArcellaResult<()> {
193
194    let (reader, mut writer) = tokio::io::split(stream);
195    let mut reader = BufReader::new(reader);
196    let mut buffer = String::new();
197
198    let result = loop {
199        buffer.clear();
200
201        let line = tokio::select! {
202            reader_result = timeout(TokioDuration::from_secs(MAX_READER_TIMEOUT), reader.read_line(&mut buffer)) => {
203                match reader_result {
204                    Ok(Ok(0)) => {
205                        tracing::trace!("Get EOF from client");
206                        break Ok(()); // EOF - client close connection
207                    },
208                    Ok(Ok(n)) => {
209                        if n > MAX_REQUEST_LENGTH {
210                            let message = format!("Request too large");
211                            let resp = AlmeResponse::error(&message);
212                            tracing::warn!("{}", message);
213                            send_response(&mut writer, &resp).await?;
214                            continue;
215                        }
216                        let trimmed = buffer.trim_end_matches(&['\r', '\n']).trim();
217                        if trimmed.is_empty() {
218                            continue;
219                        }
220                        trimmed.to_string()
221                    },
222                    Ok(Err(e)) => {
223                        tracing::error!("Recieved error: {}", e);
224                        return Err(ArcellaError::Io(e));
225                    },
226                    _ => {
227                        tracing::warn!("Reader timeout");
228                        let _ = writer.shutdown().await;
229                        tracing::debug!("Writer shutdown complete");
230                        return Ok(());
231                    }
232                }
233            },
234            _ = shutdown_rx.recv() => {
235                tracing::debug!("Connection handler received shutdown signal");
236                let _ = writer.shutdown().await;
237                tracing::debug!("Writer shutdown complete");
238                return Ok(());
239            },
240        };
241
242        let request: AlmeRequest = match serde_json::from_str(&line) {
243            Ok(req) => req,
244            Err(e) => {
245                let message = format!("Invalid JSON: {} ", e);
246                let resp = AlmeResponse::error(&message);
247                tracing::debug!("{}", message);
248                send_response(&mut writer, &resp).await?;
249                continue;
250            }
251        };
252        tracing::trace!("Get request: {:?}", request);
253
254        let response = super::commands::dispatch_command(&request.cmd, &request.args, &runtime).await;
255
256        send_response(&mut writer, &response).await?;
257
258    };
259
260    result
261
262}

264/// Serializes an [`AlmeResponse`] to JSON and writes it to the client stream.
265///
266/// A newline (`\n`) is appended to ensure line-oriented parsing on the client side.
267/// If the write fails (e.g., because the client disconnected), the error is returned
268/// so the connection handler can terminate gracefully.
269/// 
270/// # Arguments
271///
272/// * `stream` - The writable half of the Unix stream to send the response to.
273/// * `response` - The response object to serialize and send.
274async fn send_response(
275    stream: &mut WriteHalf<UnixStream>,
276    response: &AlmeResponse,
277) -> ArcellaResult<()> {
278    tracing::trace!("Send response");
279    let mut json = serde_json::to_vec(response)
280        .map_err(|e| ArcellaError::Json(e))?;
281    json.push(b'\n');
282    let _ = stream.write_all(&json).await.map_err(|e| {
283        tracing::error!("Failed to send response: {}", e);
284        ArcellaError::Io(e)
285    });
286    Ok(())
287}
