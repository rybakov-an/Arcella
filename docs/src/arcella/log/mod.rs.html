10//! Logging and tracing for the Arcella Runtime.
11//!
12//! This module implements a flexible, multi-channel logging system based on the [`tracing`] crate.
13//! It supports three output channels:
14//! - **File** (`arcella.log`) — with optional structured (JSON) or plain-text formatting;
15//! - **stderr** — for convenient debugging when running in foreground mode;
16//! - **In-memory ring buffer** — to expose recent logs via ALME (e.g., through the CLI).
17//!
18//! Configuration is read from `tracing.cfg` in Arcella’s config directory and allows:
19//! - Setting a global log level;
20//! - Configuring per-module or per-target log levels;
21//! - Enabling/disabling individual output channels;
22//! - Limiting the in-memory buffer size for ALME.
23//!
24//! The system is thread-safe and uses non-blocking I/O for file writes.
25

48/// In-memory ring buffer storing the most recent log entries.
49/// Used to serve logs via ALME (e.g., for CLI queries).
50// TODO: For high-load tasks, it is necessary to implement 
51// a lock-free circular buffer with multi-reader snapshots
52static LOG_BUFFER: std::sync::OnceLock<Arc<Mutex<VecDeque<String>>>> = std::sync::OnceLock::new();
53
54fn get_log_buffer() -> Option<&'static Arc<Mutex<VecDeque<String>>>> {
55    LOG_BUFFER.get()
56}

58/// Initializes the global `tracing` subscriber based on the provided configuration.
59///
60/// This function must be called exactly once during daemon startup. It:
61/// - Creates the log directory if it doesn’t exist;
62/// - Loads or falls back to default settings from `tracing.cfg`;
63/// - Configures logging layers: file, stderr, and in-memory buffer for ALME;
64/// - Installs a global subscriber for `tracing`.
65///
66/// # Arguments
67///
68/// * `config` — reference to the main Arcella configuration, which includes paths to logs and config files.
69///
70/// # Returns
71///
72/// A `WorkerGuard` from `tracing_appender`, which must be kept alive until shutdown
73/// to ensure buffered log entries are flushed to disk. Returns `None` if file logging is disabled.
74											  
75///
76/// # Errors
77///
78/// Returns an error if:
79/// - The log directory cannot be created;
80/// - `tracing.cfg` is malformed or contains invalid values;
81/// - The global subscriber has already been initialized;
82/// - The ALME in-memory buffer fails to initialize (when enabled).
83pub fn init(config: &ArcellaConfig) -> ArcellaResult<Option<tracing_appender::non_blocking::WorkerGuard>> {
84
85    let mut file_guard: Option<tracing_appender::non_blocking::WorkerGuard> = None;
86
87    let tracing_cfg_path = config.config_dir.join("tracing.cfg");
88
89    let tracing_cfg = load_tracing_config(&tracing_cfg_path)?;
90
91    // Ensure log directory exists
92    fs::create_dir_all(&config.log_dir)
93        .map_err(|e| ArcellaError::Io(e))?;
94
95    let log_file_path = config.log_dir.join("arcella.log");
96
97    // Initialize ALME in-memory buffer
98    if tracing_cfg.alme_buffer_size > 0 {
99        let buffer = Arc::new(Mutex::new(VecDeque::with_capacity(tracing_cfg.alme_buffer_size)));
100        LOG_BUFFER.set(buffer).map_err(|_| ArcellaError::Internal("LOG_BUFFER already set".into()))?;
101    }
102
103    // Build filter directives
104    let mut directives = vec![format!("arcella={}", tracing_cfg.default_level)];
105
106    // Override per-module levels
107    for (target, level) in &tracing_cfg.modules {
108        directives.push(format!("{}={}", target, level));
109    }
110
111    let filter = directives.join(",");
112
113    let env_filter = EnvFilter::try_new(filter)
114        .map_err(|e| ArcellaError::Config(format!("invalid log filter: {}", e)))?;
115
116
117    let mut layers = Vec::new();
118
119    // 1. File layer
120    if tracing_cfg.file {
121
122        // Use `never` rolling (single file: arcella.log)
123        let file_appender = tracing_appender::rolling::never(&config.log_dir, "arcella.log");
124        let (non_blocking, guard) = tracing_appender::non_blocking(file_appender);
125
126        let file_layer = if tracing_cfg.structured {
127            fmt::layer()
128                .json()
129                .with_writer(non_blocking)
130                .with_ansi(false)
131                .boxed()
132        } else {
133            fmt::layer()
134                .with_writer(non_blocking)
135                .with_ansi(false)
136                .boxed()
137        };
138        layers.push(file_layer);
139        file_guard = Some(guard);
140    }
141
142    // 2. StdErr layer
143    if tracing_cfg.stderr {
144        let console_layer = fmt::layer()
145            .with_writer(std::io::stderr)
146            .with_ansi(true)
147            .boxed();
148        layers.push(console_layer);
149    }
150
151    // 3. In-memory ALME layer
152    if tracing_cfg.alme_buffer_size > 0 {
153        let alme_layer = AlmeBufferLayer::new(tracing_cfg.alme_buffer_size);
154        layers.push(Box::new(alme_layer));
155    }    
156
157    let subscriber = tracing_subscriber::registry()
158        .with(layers)
159        .with(env_filter);
160
161    subscriber
162        .try_init()
163        .map_err(|e| ArcellaError::Internal(format!("failed to init tracing: {}", e)))?;
164
165
166    Ok(file_guard)
167}

169/// Loads tracing configuration from a TOML file.
170///
171/// If the file does not exist, default settings are returned.
172fn load_tracing_config(path: &PathBuf) -> ArcellaResult<TracingConfig> {
173    if !path.exists() {
174        tracing::debug!("tracing.cfg not found, using defaults");
175        return Ok(TracingConfig::default());
176    }
177
178    let contents = fs::read_to_string(path)
179        .map_err(|e| ArcellaError::IoWithPath{source: e, path: path.clone()})?;
180    toml::from_str(&contents)
181        .map_err(|e| ArcellaError::Config(format!("tracing.cfg: {}", e)))
182}

185/// Helper: deserialize LevelFilter from string (e.g., "info", "debug")
186fn deserialize_level_filter<'de, D>(deserializer: D) -> Result<LevelFilter, D::Error>
187where
188    D: Deserializer<'de>,
189{
190    let s = String::deserialize(deserializer)?;
191    s.parse::<LevelFilter>().map_err(serde::de::Error::custom)
192}
193
194/// Helper: deserialize a map of per-module log levels.
195fn deserialize_module_levels<'de, D>(deserializer: D) -> Result<HashMap<String, LevelFilter>, D::Error>
196where
197    D: Deserializer<'de>,
198{
199    let map: HashMap<String, String> = Deserialize::deserialize(deserializer)?;
200    let mut result = HashMap::new();
201    for (target, level_str) in map {
202        let level = level_str
203            .parse::<LevelFilter>()
204            .map_err(serde::de::Error::custom)?;
205        result.insert(target, level);
206    }
207    Ok(result)
208}

210/// Arcella tracing configuration.
211#[derive(Deserialize, Debug, Clone)]
212pub struct TracingConfig {
213    /// Default log level for targets under the `arcella` namespace.
214    ///
215    /// Valid values: `"trace"`, `"debug"`, `"info"`, `"warn"`, `"error"`.
216    #[serde(default = "default_log_level", deserialize_with = "deserialize_level_filter")]
217    pub default_level: LevelFilter,
218
219    /// Use structured (JSON) format for file logging.
220    ///
221    /// If `true`, entries in `arcella.log` will be JSON-encoded, suitable for machine processing
222    /// (e.g., ingestion into ELK or Loki). If `false`, human-readable text is used.
223    #[serde(default = "default_structured")]
224    pub structured: bool,
225
226    /// Output logs to stderr.
227    ///
228    /// Useful when running Arcella in foreground mode or inside a container where stderr is
229    /// captured by an orchestrator (e.g., systemd or Kubernetes).
230    #[serde(default = "default_stderr")]
231    pub stderr: bool,
232
233    /// Write logs to the `arcella.log` file.
234    ///
235    /// The file is created in the directory specified by `ArcellaConfig::log_dir`.
236    #[serde(default = "default_file")]
237    pub file: bool,
238
239    /// Maximum size of the in-memory ring buffer for ALME (in log entries).
240    ///
241    /// A value of `0` disables the buffer. Used by the `alme logs` command to retrieve
242    /// the most recent `N` lines without reading the log file.
243    #[serde(default = "default_alme_buffer_size")]
244    pub alme_buffer_size: usize,
245
246    /// Per-module/target log levels.
247    ///
248    /// Keys are target names (e.g., `"arcella::runtime"`), values are log levels.
249    /// These override `default_level` for the specified targets.
250    ///
251    /// Example:
252    /// ```toml
253    /// [modules]
254    /// "arcella::runtime" = "info"
255    /// "arcella::alme" = "debug"
256    /// ```
257    #[serde(default, deserialize_with = "deserialize_module_levels")]
258    pub modules: HashMap<String, LevelFilter>,
259}
260
261fn default_log_level() -> LevelFilter { LevelFilter::INFO }
262fn default_structured() -> bool { false }
263fn default_stderr() -> bool { true }
264fn default_file() -> bool { true }
265fn default_alme_buffer_size() -> usize { 100 }
266
267impl Default for TracingConfig {
268    fn default() -> Self {
269        Self {
270            default_level: default_log_level(),
271            structured: default_structured(),
272            stderr: default_stderr(),
273            file: default_file(),
274            alme_buffer_size: default_alme_buffer_size(),
275            modules: HashMap::new(),
276        }
277    }
278}

282/// A tracing layer that writes events to an in-memory ring buffer for later retrieval via ALME.
283struct AlmeBufferLayer {
284    max_size: usize,
285}
286
287impl AlmeBufferLayer {
288    fn new(max_size: usize) -> Self {
289        Self { max_size }
290    }
291}
292
293impl<S> Layer<S> for AlmeBufferLayer
294where
295    S: tracing::Subscriber + for<'a> LookupSpan<'a>,
296{
297    fn on_event(
298        &self,
299        event: &tracing::Event<'_>,
300        _ctx: tracing_subscriber::layer::Context<'_, S>,
301    ) {
302        if let Some(buffer) = get_log_buffer() {
303            let meta = event.metadata();
304
305            let now: OffsetDateTime = OffsetDateTime::now_utc();
306            let now_rfc3339 = now.format(&time::format_description::well_known::Rfc3339)
307                .unwrap_or_else(|_| "<invalid-timestamp>".to_string());
308
309            let mut visitor = EventVisitor::default();
310            event.record(&mut visitor);
311
312            let message = if visitor.message.is_empty() {
313                "no message".to_string()
314            } else {
315                visitor.message
316            };
317
318            let fields = if visitor.fields.is_empty() {
319                String::new()
320            } else {
321                format!(" {{{}}}", visitor.fields.join(", "))
322            };
323
324            let line = format!(
325                "{} {} {}: {}{}",
326                now_rfc3339,
327                meta.level(),
328                meta.target(),
329                message,
330                fields
331            ); 
332
333            match buffer.lock() {
334                Ok(mut buf) => {
335                    if buf.len() >= self.max_size {
336                        buf.pop_front();
337                    }
338                    buf.push_back(line);
339                }
340                Err(e) => {
341                    // Avoid panicking in a tracing handler; silently ignore if poisoned
342                    eprintln!("ALME log buffer poisoned: {}", e);
343                }
344            }
345
346        }
347    }
348}

350/// Returns up to `n` most recent log entries from the in-memory buffer.
351///
352/// The buffer is only populated if `alme_buffer_size > 0` in `tracing.cfg`.
353/// Entries are returned in reverse chronological order (most recent first).
354///
355/// # Arguments
356///
357/// * `n` — maximum number of log lines to return.
358///
359/// # Returns
360///
361/// A vector of log strings. Returns an empty vector if the buffer is uninitialized or disabled.
362pub fn get_recent_logs(n: usize) -> Vec<String> {
363    if let Some(buffer) = get_log_buffer() {
364        match buffer.lock() {
365            Ok(buf) => {
366                buf.iter().rev().take(n).cloned().collect()
367            }
368            Err(e) => {
369                eprintln!("Failed to lock ALME log buffer: {}", e);
370                vec![]
371            }
372        }
373    } else {
374        vec![]
375    }
376}

378#[derive(Default)]
379struct EventVisitor {
380    message: String,
381    fields: Vec<String>,
382}
383
384impl tracing::field::Visit for EventVisitor {
385    fn record_str(&mut self, field: &tracing::field::Field, value: &str) {
386        if field.name() == "message" {
387            self.message = value.to_string();
388        } else {
389            self.fields.push(format!("{}={}", field.name(), value));
390        }
391    }
392
393    fn record_debug(&mut self, field: &tracing::field::Field, value: &dyn std::fmt::Debug) {
394        if field.name() == "message" {
395            self.message = format!("{:?}", value);
396        } else {
397            self.fields.push(format!("{}={:?}", field.name(), value));
398        }
399    }
