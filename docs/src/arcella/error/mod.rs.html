10//! Centralized error handling for Arcella.
11//!
12//! Uses `thiserror` to define structured errors and `anyhow` for convenient propagation.
13//! All modules should return `Result<T, ArcellaError>` for internal logic,
14//! and `anyhow::Result<T>` (aliased as `Result<T>`) for top-level functions like `main`.
15//! 
16
17use std::path::PathBuf;
18use thiserror::Error;
19use tokio::task::JoinError;
20
21use arcella_wasmtime::error::ArcellaWasmtimeError;
22
23/// The root error type for all Arcella-specific failures.
24#[derive(Error, Debug)]
25pub enum ArcellaError {
26    /// General-purpose error for unexpected conditions.
27    #[error("Internal error: {0}")]
28    Internal(String),
29
30    /// I/O error (file not found, permission denied, etc.).
31    #[error("I/O error: {0}")]
32    Io(#[from] std::io::Error),
33
34    /// IO error with associated path for better diagnostics
35    #[error("I/O error at {path:?}: {source}")]
36    IoWithPath {
37        source: std::io::Error,
38        path: PathBuf,
39    },
40
41    /// Failed to parse WebAssembly Text Format (`.wat`).
42    #[error("WAT parsing error: {0}")]
43    Wat(#[from] wat::Error),
44
45    /// Configuration loading or parsing error.
46    #[error("Config error: {0}")]
47    Config(String),
48
49    /// JSON serialization/deserialization error.
50    #[error("JSON error: {0}")]
51    Json(#[from] serde_json::Error), 
52
53    /// Task join error.
54    #[error("Task join error: {0}")]
55    Join(#[from] JoinError),
56
57    /// Runtime error.
58    #[error("Runtime error: {0}")]
59    RuntimeError(String),
60
61    #[error("Arcella Wasmtime error: {0}")]
62    ArcellaWasmtimeError (#[from] ArcellaWasmtimeError),    
63
64}
65
66/// Convenient alias for `Result<T, ArcellaError>`.
67///
68/// Use this in internal module APIs (e.g., `runtime::install_module`).
69pub type Result<T> = std::result::Result<T, ArcellaError>;
70
71// Re-export `anyhow::Result` as `AnyResult` for top-level use (optional but clean)
72// Alternatively, you can use `anyhow::Result` directly in `main.rs`
73pub use anyhow::Result as AnyResult;
