10use std::path::PathBuf;
11use serde::Deserialize;
12
13use crate::error::{ArcellaError, Result as ArcellaResult};
14
15#[derive(Debug, Clone, Deserialize)]
16pub struct ArcellaConfig {
17    pub base_dir: PathBuf,
18    pub config_dir: PathBuf,
19    pub log_dir: PathBuf,
20    pub modules_dir: PathBuf,
21    pub cache_dir: PathBuf,
22    pub socket_path: PathBuf,
23}
24
25impl Default for ArcellaConfig {
26    fn default() -> Self {
27        let base = dirs::home_dir().unwrap().join(".arcella");
28        Self {
29            base_dir: base.clone(),
30            config_dir: base.join("config"),
31            log_dir: base.join("log"),
32            modules_dir: base.join("modules"),
33            cache_dir: base.join("cache"),
34            socket_path: base.join("alme"),
35        }
36    }
37
38}
39
40pub async fn load() -> ArcellaResult<ArcellaConfig> {
41    // Пока просто возвращаем default
42    Ok(ArcellaConfig::default())
43}
