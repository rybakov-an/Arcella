73#[derive(Debug, Clone)]
74pub struct IntegrityChecker {
75    paths: Vec<PathBuf>,
76    initial_mtimes: HashMap<PathBuf, SystemTime>,
77}
78
79impl IntegrityChecker {
80    pub fn new(paths: Vec<PathBuf>) -> ArcellaResult<Self> {
81        let mut initial_mtimes = HashMap::new();
82        for path in &paths {
83            let metadata = std::fs::metadata(path)
84                .map_err(|e| ArcellaError::IoWithPath { source: e, path: path.clone() })?;
85            let mtime = metadata.modified()
86                .map_err(|e| ArcellaError::Internal(format!("Cannot get mtime for {:?}: {}", path, e)))?;
87            initial_mtimes.insert(path.clone(), mtime);
88        }
89        Ok(IntegrityChecker { paths, initial_mtimes })
90    }
91
92    pub async fn check(&self) -> ArcellaResult<()> {
93        let current_mtimes = get_current_mtimes(&self.paths).await?;
94        check_mtimes_changed(&self.initial_mtimes, &current_mtimes)
95    }
96}
97
98fn check_mtimes_changed(
99    initial_mtimes: &HashMap<PathBuf, std::time::SystemTime>,
100    current_mtimes: &HashMap<PathBuf, std::time::SystemTime>,
101) -> ArcellaResult<()> {
102    for (path, current_mtime) in current_mtimes {
103        if let Some(initial_mtime) = initial_mtimes.get(path) {
104            if current_mtime != initial_mtime {
105                return Err(ArcellaError::Internal(
106                    format!("Config integrity violation: file {:?} was modified after startup", path)
107                ));
108            }
109        } else {
110            return Err(ArcellaError::Internal(
111                format!("Config integrity violation: file {:?} not found in initial list", path)
112            ));
113        }
114    }
115    Ok(())
116}

118async fn get_current_mtimes(paths: &[PathBuf]) -> ArcellaResult<HashMap<PathBuf, std::time::SystemTime>> {
119    // Создадим вектор future'ов для каждой проверки mtime
120    let checks: Vec<_> = paths.iter().map(|path| {
121        let path = path.clone();
122        async move {
123            let metadata = tokio::fs::metadata(&path).await
124                .map_err(|e| ArcellaError::IoWithPath { source: e, path: path.clone() })?;
125            let mtime = metadata.modified()
126                .map_err(|e| ArcellaError::Internal(format!("Cannot get mtime for {:?}: {}", path, e)))?;
127            Ok::<(PathBuf, std::time::SystemTime), ArcellaError>((path, mtime))
128        }
129    }).collect();
130
131    // Запускаем все future'ы параллельно и ждем их завершения
132    let results = future::join_all(checks).await;
133
134    let mut current_mtimes = HashMap::with_capacity(results.len());
135    for result in results {
136        let (path, mtime) = result?; // Если одна из проверок завершится с ошибкой, propagate
137        current_mtimes.insert(path, mtime);
138    }
139
140    Ok(current_mtimes)
141}

202pub fn collect_paths_recursive(
203    item: &TomlItem,
204    current_path: &[String],
205    includes: &mut Vec<String>,
206    values: &mut HashMap<String, TValue>,
207    depth: usize,
208    max_depth: usize,
209) -> ArcellaResult<()> {
210    if depth > max_depth {
211        return Ok(());
212    }
213
214    match item {
215        TomlItem::Table(table) => {
216            for (key, value) in table {
217                let mut key_path = current_path.to_vec();
218                key_path.push(key.into());
219
220                if key == "includes" {
221                    match value {
222                        TomlItem::Value(TomlValue::Array(includes_array)) => {
223                            for include in includes_array {
224                                if let Some(str_val) = include.as_str() {
225                                    includes.push(str_val.to_owned());
226                                }
227                            }
228                        },
229                        // Also handle a single string value for 'includes'
230                        TomlItem::Value(include) => {
231                            if let Some(str_val) = include.as_str() {
232                                includes.push(str_val.to_owned());
233                            }
234                        },
235                        _ => {} 
236                    };
237                } else if let TomlItem::Value(subvalue) = value {
238                    values.insert(
239                        key_path.join("."), 
240                        TValue::from_toml_value(subvalue)?
241                    );
242                } else {
243                    collect_paths_recursive(
244                        value,
245                        &key_path,
246                        includes,
247                        values,
248                        depth + 1,
249                        max_depth,
250                    )?;                    
251                }
252            }
253        },
254        _ => {}
255    }        
256
257    Ok(())
258}
